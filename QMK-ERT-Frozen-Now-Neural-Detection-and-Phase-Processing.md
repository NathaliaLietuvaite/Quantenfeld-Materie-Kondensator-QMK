## **APPENDIX A: FPGA-VERILOG FÜR DETECTION**

```verilog
module Frozen_Now_Detector (
    input clk, rst,
    input [31:0] neural_signal_real,  // I-Kanal
    input [31:0] neural_signal_imag,  // Q-Kanal
    output reg [15:0] phase_delta,  // ΔΦ
    output reg detect_trigger
);
    parameter THRESHOLD = 16'h0010;  // 0.01 Q8.8
    
    reg [31:0] phase_calc;
    
    always @(posedge clk or posedge rst) begin
        if (rst) begin
            detect_trigger <= 1'b0;
            phase_delta <= 16'h0;
        end else begin
            phase_calc = $atan2(neural_signal_imag, neural_signal_real);  // Phase-Extraktion
            phase_delta = phase_calc[15:0];  // Proxy für ΔΦ
            detect_trigger <= (phase_delta > THRESHOLD);
        end
    end
endmodule
```

## **APPENDIX B: PYTHON-STEUEREINHEIT FÜR MATRIX-BAU**

```python
import numpy as np
import time

class Frozen_Now_Controller:
    def __init__(self, fpga_interface, odos_guardian, fidelity_threshold=0.95):
        self.fpga = fpga_interface
        self.odos = odos_guardian
        self.matrix = []  # Liste von Frozen Nows
        
    def capture_frozen_now(self, neural_data):
        """Erfasst und prozessiert Signal."""
        real, imag = neural_data['real'], neural_data['imag']
        phase = np.arctan2(imag, real)
        amplitude = np.sqrt(real**2 + imag**2)
        state_vector = amplitude * np.exp(1j * phase)
        fidelity, delta_e = self.validate(state_vector)
        if fidelity > fidelity_threshold and delta_e < 0.05:
            self.matrix.append(state_vector)
            return True, state_vector
        return False, None
    
    def validate(self, vector):
        """Fidelity & ΔE (simuliert)."""
        target = np.exp(1j * np.pi / 4)  # Ideal-Proxy
        fidelity = np.abs(np.dot(vector.conj(), target)) ** 2
        entropy = -np.sum(np.abs(vector)**2 * np.log(np.abs(vector)**2 + 1e-10))
        return fidelity, entropy / np.log(2)
    
    def build_matrix(self):
        """Baut Matrix aus Frozen Nows."""
        if len(self.matrix) > 1:
            return np.outer(self.matrix[0], self.matrix[1].conj())  # Beispiel-Matrix
        return None
    
    def realtime_loop(self):
        while True:
            data = self.fpga.read_neural_data()  # Hypothetisch
            success, vec = self.capture_frozen_now(data)
            if success:
                matrix = self.build_matrix()
                if matrix is not None:
                    self.odos.log_matrix(matrix)
            time.sleep(0.001)  # 1kHz
```
